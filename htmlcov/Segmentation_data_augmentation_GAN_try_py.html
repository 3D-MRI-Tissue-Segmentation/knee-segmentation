85def discriminator_loss(real_ouput, fake_output): 

86 real_loss = cross_entropy(tf.ones_like(real_ouput), real_ouput) 

87 fake_loss = cross_entropy(tf.zeros_like(fake_output), fake_output) 

88 total_loss = real_loss + fake_loss
